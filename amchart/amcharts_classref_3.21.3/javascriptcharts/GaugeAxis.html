Creates an axis for AmAngularGauge charts, multiple can be assigned.

Example

var chart = AmCharts.makeChart("chartdiv",{
  ...
  "axes": [
    {
      "startValue": 0,
      "endValue": 100,
      "valueInterval": 10,
      "bands": [
        {
          "startValue": 0,
          "endValue": 100
        }
      ]
    }
  ]
});
